  <script>

    const STORAGE_KEY = 'baustellenkontrolle_singlefile_v1';

    const state = loadState() || {

      projects: [

        {

          id: 'p1',

          name: 'Wohnanlage Musterstraße',

          location: 'Hamburg',

          favorite: true,

          structures: [

            {

              id: 's1', name: 'Gebäude A', inspections: [

                { id:'i1', date:'2026-04-17', inspector:'Michel', points:[

                  { id:'pt1', component:'Fundament F5', aspect:'untere Bewehrung', status:'Mangel', note:'zu große Abstände', photos:1, createdAt:'09:10' },

                  { id:'pt2', component:'Fundament F5', aspect:'obere Bewehrung', status:'OK', note:'', photos:0, createdAt:'09:14' },

                  { id:'pt3', component:'Wand W2', aspect:'Schubbewehrung', status:'Mangel', note:'fehlt unterhalb der Wand', photos:2, createdAt:'09:22' }

                ]}

              ]

            },

            { id:'s2', name:'Gebäude B', inspections:[] },

            { id:'s3', name:'Tiefgarage', inspections:[] }

          ]

        },

        { id:'p2', name:'Logistikhalle Nord', location:'Bremen', favorite:true, structures:[{ id:'s4', name:'Halle 1', inspections:[] }] },

        { id:'p3', name:'Parkhaus Innenstadt', location:'Kiel', favorite:false, structures:[{ id:'s5', name:'Bauwerk Ost', inspections:[] }, { id:'s6', name:'Bauwerk West', inspections:[] }] }

      ],

      selectedProjectId: 'p1',

      selectedStructureId: 's1',

      selectedInspectionId: 'i1',

      tab: 'projekte',

      search: '',

      favoritesOnly: false,

      editingPointId: null,

      isMobileNavOpen: true,

      dialogs: { project:false, structure:false },

      projectForm: { id:null, name:'', location:'', firstStructure:'' },

      structureForm: { name:'' },

      form: { component:'', aspect:'', status:'OK', note:'', photoAttached:false, photoCount:0, photoNames:[] }

    };


    const STANDARD_COMPONENTS = ['Fundament','Einzelfundament','Streifenfundament','Bodenplatte','Wand','Stütze','Unterzug','Überzug','Decke','Treppe','Balkonplatte','Attika','Ringbalken','Ringanker','Sturz','Binder','Pfette','Mauerwerk','Fertigteilwand','Fertigteildecke'];

    const ASPECT_SUGGESTIONS = ['untere Bewehrung','obere Bewehrung','Schubbewehrung','Anschlussbewehrung','Bügel','Betondeckung','Durchstanzbewehrung','Schalung','Abmessungen','Einbauteile'];


    function loadState() { try { return JSON.parse(localStorage.getItem(STORAGE_KEY)); } catch { return null; } }

    function saveState() { localStorage.setItem(STORAGE_KEY, JSON.stringify(state)); }

    function todayIso() { return new Date().toISOString().slice(0,10); }

    function nowTime() { return new Date().toLocaleTimeString('de-DE',{hour:'2-digit', minute:'2-digit'}); }

    function formatDate(dateStr) { return new Date(dateStr).toLocaleDateString('de-DE',{year:'numeric', month:'2-digit', day:'2-digit'}); }

    function cls(...arr){ return arr.filter(Boolean).join(' '); }

    function selectedProject(){ return state.projects.find(p => p.id === state.selectedProjectId) || state.projects[0]; }

    function selectedStructure(){ const p = selectedProject(); return p?.structures.find(s => s.id === state.selectedStructureId) || p?.structures[0]; }

    function selectedInspection(){ const s = selectedStructure(); return s?.inspections.find(i => i.id === state.selectedInspectionId) || s?.inspections[0]; }

    function icon(name){ return name; }

    function filteredProjects(){ const q = state.search.toLowerCase(); return state.projects.filter(project => {

      const fav = state.favoritesOnly ? project.favorite : true;

      const match = !q || project.name.toLowerCase().includes(q) || (project.location || '').toLowerCase().includes(q) || project.structures.some(s => s.name.toLowerCase().includes(q));

      return fav && match;

    }); }

    function componentSuggestions(){ const set = new Set(STANDARD_COMPONENTS); const p = selectedProject(); if (p) p.structures.forEach(s => s.inspections.forEach(i => i.points.forEach(pt => pt.component && set.add(pt.component)))); const q = state.form.component.trim().toLowerCase(); const arr = Array.from(set); return (!q ? arr : arr.filter(x => x.toLowerCase().includes(q))).slice(0,8); }

    function aspectSuggestions(){ const q = state.form.aspect.trim().toLowerCase(); return (!q ? ASPECT_SUGGESTIONS : ASPECT_SUGGESTIONS.filter(x => x.toLowerCase().includes(q))).slice(0,6); }

    function ensureSelection(){

      const p = selectedProject();

      if (!p) return;

      if (!state.projects.some(x => x.id === state.selectedProjectId)) {

        state.selectedProjectId = state.projects[0]?.id || '';

      }

      const p2 = selectedProject();

      if (p2 && !p2.structures.some(x => x.id === state.selectedStructureId)) state.selectedStructureId = p2.structures[0]?.id || '';

      const s = selectedStructure();

      if (s && !s.inspections.some(x => x.id === state.selectedInspectionId)) state.selectedInspectionId = s.inspections[0]?.id || '';

    }

    function update(mutator){ mutator(); ensureSelection(); saveState(); render(); }

    function resetPointForm(keepComponent=false){ state.form = { component: keepComponent ? state.form.component : '', aspect:'', status:'OK', note:'', photoAttached:false, photoCount:0, photoNames:[] }; state.editingPointId = null; }

    function openProjectDialog(editProject){ state.projectForm = editProject ? { id:editProject.id, name:editProject.name, location:editProject.location || '', firstStructure:'' } : { id:null, name:'', location:'', firstStructure:'' }; state.dialogs.project = true; render(); }

    function openStructureDialog(){ state.structureForm = { name:'' }; state.dialogs.structure = true; render(); }

    function selectProjectAction(project){ update(() => { state.selectedProjectId = project.id; state.selectedStructureId = project.structures[0]?.id || ''; state.selectedInspectionId = project.structures[0]?.inspections[0]?.id || ''; state.tab = 'bauwerke'; }); }

    function selectStructureAction(structure){ update(() => { state.selectedStructureId = structure.id; state.selectedInspectionId = structure.inspections[0]?.id || ''; state.tab = 'kontrollen'; }); }

    function createProject(){ if (!state.projectForm.name.trim()) return; update(() => {

      if (state.projectForm.id) {

        state.projects = state.projects.map(project => project.id !== state.projectForm.id ? project : { ...project, name: state.projectForm.name.trim(), location: state.projectForm.location.trim() });

      } else {

        const stamp = Date.now();

        const newProject = { id:`p${stamp}`, name:state.projectForm.name.trim(), location:state.projectForm.location.trim(), favorite:false, structures:[{ id:`s${stamp}`, name: state.projectForm.firstStructure.trim() || 'Bauwerk 1', inspections:[] }] };

        state.projects.unshift(newProject);

        state.selectedProjectId = newProject.id;

        state.selectedStructureId = newProject.structures[0].id;

        state.selectedInspectionId = '';

        state.tab = 'bauwerke';

      }

      state.dialogs.project = false;

      state.projectForm = { id:null, name:'', location:'', firstStructure:'' };

    }); }

    function createStructure(){ const p = selectedProject(); if (!p || !state.structureForm.name.trim()) return; update(() => { const stamp = Date.now(); const newStructure = { id:`s${stamp}`, name:state.structureForm.name.trim(), inspections:[] }; p.structures.push(newStructure); state.selectedStructureId = newStructure.id; state.selectedInspectionId=''; state.tab='bauwerke'; state.dialogs.structure=false; }); }

    function createInspection(){ const s = selectedStructure(); if (!s) return; update(() => { const newInspection = { id:`i${Date.now()}`, date:todayIso(), inspector:'Michel', points:[] }; s.inspections.unshift(newInspection); state.selectedInspectionId = newInspection.id; state.tab = 'kontrollen'; }); }

    function deleteInspection(id){ const s = selectedStructure(); if (!s) return; update(() => { s.inspections = s.inspections.filter(i => i.id !== id); if (state.selectedInspectionId === id) state.selectedInspectionId = s.inspections[0]?.id || ''; }); }

    function deleteProject(id){ update(() => { state.projects = state.projects.filter(p => p.id !== id); const p = state.projects[0]; if (p) { state.selectedProjectId = p.id; state.selectedStructureId = p.structures[0]?.id || ''; state.selectedInspectionId = p.structures[0]?.inspections[0]?.id || ''; } }); }

    function toggleFavorite(id){ update(() => { const p = state.projects.find(x => x.id === id); if (p) p.favorite = !p.favorite; }); }

    function savePoint(opts={ keepComponent:false, closeAfter:false }){ const inspection = selectedInspection(); if (!inspection || !state.form.component.trim()) return; update(() => {

      const payload = { component:state.form.component.trim(), aspect:state.form.aspect.trim(), status:state.form.status, note:state.form.note.trim(), photos: state.form.photoCount || (state.form.photoAttached ? 1 : 0), photoNames: state.form.photoNames || [] };

      if (state.editingPointId) {

        inspection.points = inspection.points.map(p => p.id !== state.editingPointId ? p : { ...p, ...payload });

      } else {

        inspection.points.unshift({ id:`pt${Date.now()}`, ...payload, createdAt: nowTime() });

      }

      resetPointForm(opts.keepComponent);

      if (opts.closeAfter) state.tab = 'kontrollen';

    }); }

    function editPoint(id){ const inspection = selectedInspection(); const point = inspection?.points.find(p => p.id === id); if (!point) return; state.editingPointId = id; state.form = { component:point.component, aspect:point.aspect, status:point.status, note:point.note, photoAttached:point.photos>0, photoCount:point.photos||0, photoNames:point.photoNames||[] }; state.tab='neu'; saveState(); render(); }

    function deletePoint(id){ const inspection = selectedInspection(); if (!inspection) return; update(() => { inspection.points = inspection.points.filter(p => p.id !== id); if (state.editingPointId === id) resetPointForm(false); }); }

    function groupedPoints(){ const inspection = selectedInspection(); const map = new Map(); (inspection?.points || []).forEach(point => { const key = point.component.trim() || 'Ohne Bauteil'; if (!map.has(key)) map.set(key, []); map.get(key).push(point); }); return Array.from(map.entries()); }


    function render(){

      ensureSelection();

      const app = document.getElementById('app');

      const project = selectedProject();

      const structure = selectedStructure();

      const inspection = selectedInspection();

      app.innerHTML = `

        <div class="app">

          <div class="topbar">

            <div>

              <h1 class="title">Baustellenkontrolle</h1>

              <p class="subtitle">Prototyp für die mobile Erfassung statisch-konstruktiver Kontrollen</p>

            </div>

            <div class="row wrap">

              <span class="badge">${escapeHtml(project?.name || '')}</span>

              <span class="badge">${escapeHtml(structure?.name || '')}</span>

              <button class="btn btn-outline mobile-inline" data-action="toggle-nav">${state.isMobileNavOpen ? 'Navigation schließen' : 'Navigation öffnen'}</button>

            </div>

          </div>


          <div class="layout">

            <div class="card ${!state.isMobileNavOpen ? 'mobile-only hidden' : ''}">

              <div class="card-header"><div class="card-title">Navigation</div></div>

              <div class="card-content stack-md">

                <div class="tabs desktop-only">

                  ${tabButton('projekte','Projekte')}

                  ${tabButton('bauwerke','Bauwerke')}

                  ${tabButton('kontrollen','Kontrollen')}

                </div>

                ${state.tab === 'projekte' ? projectsView() : ''}

                ${state.tab === 'bauwerke' ? structuresView(project) : ''}

                ${state.tab === 'kontrollen' ? inspectionsView(structure) : ''}

                ${state.tab === 'neu' ? pointFormView() : ''}

              </div>

            </div>


            <div class="stack-lg">

              <div class="content-grid">

                <div class="card">

                  <div class="card-header"><div class="card-title">Kontrolle ${inspection ? formatDate(inspection.date) : '–'}</div></div>

                  <div class="card-content">

                    <div class="scroll large stack-sm">${pointsView(inspection)}</div>

                  </div>

                </div>

                ${state.tab === 'kontrollen' ? exportView(project, structure, inspection) : ''}

              </div>

            </div>

          </div>

        </div>


        <div class="bottom-nav mobile-only">

          <div class="bottom-nav-grid">

            ${bottomTab('projekte','Projekte')}

            ${bottomTab('bauwerke','Bauwerke')}

            ${bottomTab('kontrollen','Kontrollen')}

          </div>

        </div>


        ${state.dialogs.project ? projectDialog() : ''}

        ${state.dialogs.structure ? structureDialog() : ''}

      `;

      bindEvents();

    }


    function tabButton(value,label){ return `<button class="tab-btn ${state.tab===value?'active':''}" data-tab="${value}">${label}</button>`; }

    function bottomTab(value,label){ return `<button class="bottom-tab ${state.tab===value?'active':''}" data-tab="${value}">${label}</button>`; }

    function projectsView(){ return `

      <div class="stack-md">

        <div class="search-wrap"><span class="search-icon">⌕</span><input class="input" placeholder="Projekt oder Bauwerk suchen" value="${escapeAttr(state.search)}" data-field="search"></div>

        <div class="grid-2">

          <button class="btn ${state.favoritesOnly ? '' : 'btn-outline'}" data-action="toggle-favorites">Nur Favoriten</button>

          <button class="btn" data-action="open-project-dialog">Neues Projekt</button>

        </div>

        <div class="scroll stack-sm">${filteredProjects().map(project => `

          <div class="card click-card ${state.selectedProjectId===project.id?'selected':''}" data-select-project="${project.id}">

            <div class="card-content row between">

              <div class="grow">

                <div><strong>${escapeHtml(project.name)}</strong></div>

                <div class="muted">${escapeHtml(project.location || '–')}</div>

                <div class="fine">${project.structures.length} Bauwerke</div>

              </div>

              <div class="actions">

                <button class="icon-btn" data-fav="${project.id}">${project.favorite ? '★' : '☆'}</button>

                <button class="icon-btn" data-edit-project="${project.id}">✎</button>

                <button class="icon-btn" data-delete-project="${project.id}">🗑</button>

              </div>

            </div>

          </div>`).join('')}</div>

      </div>`; }

    function structuresView(project){ return `

      <div class="stack-md">

        <div class="row between">

          <div class="grow"><div class="muted">Projekt</div><div class="box"><strong>${escapeHtml(project?.name || '')}</strong></div></div>

          <button class="btn" data-action="open-structure-dialog">Bauwerk</button>

        </div>

        ${(project?.structures || []).length ? project.structures.map(structure => `<button class="btn ${state.selectedStructureId===structure.id?'':'btn-outline'}" data-select-structure="${structure.id}">${escapeHtml(structure.name)}</button>`).join('') : `<div class="empty">Noch keine Bauwerke vorhanden.</div>`}

      </div>`; }

    function statusClass(status){ return status === 'OK' ? 'status-ok' : status === 'Hinweis' ? 'status-hinweis' : 'status-mangel'; }

    function escapeHtml(str){ return String(str ?? '').replace(/[&<>"']/g, s => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[s])); }

    function escapeAttr(str){ return escapeHtml(str).replace(/`/g, '&#96;'); }


    function bindEvents(){

      document.querySelectorAll('[data-tab]').forEach(el => el.onclick = () => update(() => { state.tab = el.dataset.tab; }));

      const actionMap = {

        'toggle-nav': () => update(() => state.isMobileNavOpen = !state.isMobileNavOpen),

        'toggle-favorites': () => update(() => state.favoritesOnly = !state.favoritesOnly),

        'open-project-dialog': () => openProjectDialog(null),

        'open-structure-dialog': () => openStructureDialog(),

        'create-inspection': () => createInspection(),

        'cancel-edit': () => update(() => resetPointForm(false)),

        'create-project': () => createProject(),

        'create-structure': () => createStructure(),

      };

      document.querySelectorAll('[data-action]').forEach(el => el.onclick = () => actionMap[el.dataset.action]?.());

      document.querySelectorAll('[data-select-project]').forEach(el => el.onclick = () => selectProjectAction(state.projects.find(p => p.id === el.dataset.selectProject)));

      document.querySelectorAll('[data-select-structure]').forEach(el => el.onclick = () => { const p = selectedProject(); const s = p?.structures.find(x => x.id === el.dataset.selectStructure); if (s) selectStructureAction(s); });

      document.querySelectorAll('[data-select-inspection]').forEach(el => el.onclick = () => update(() => state.selectedInspectionId = el.dataset.selectInspection));

      document.querySelectorAll('[data-delete-inspection]').forEach(el => el.onclick = () => deleteInspection(el.dataset.deleteInspection));

      document.querySelectorAll('[data-delete-project]').forEach(el => el.onclick = (e) => { e.stopPropagation(); deleteProject(el.dataset.deleteProject); } );

      document.querySelectorAll('[data-fav]').forEach(el => el.onclick = (e) => { e.stopPropagation(); toggleFavorite(el.dataset.fav); } );

      document.querySelectorAll('[data-edit-project]').forEach(el => el.onclick = (e) => { e.stopPropagation(); const p = state.projects.find(x => x.id === el.dataset.editProject); if (p) openProjectDialog(p); });

      document.querySelectorAll('[data-chip-component]').forEach(el => el.onclick = () => update(() => state.form.component = el.dataset.chipComponent));

      document.querySelectorAll('[data-chip-aspect]').forEach(el => el.onclick = () => update(() => state.form.aspect = el.dataset.chipAspect));

      document.querySelectorAll('[data-status]').forEach(el => el.onclick = () => update(() => state.form.status = el.dataset.status));

      document.querySelectorAll('[data-edit-point]').forEach(el => el.onclick = () => editPoint(el.dataset.editPoint));

      document.querySelectorAll('[data-delete-point]').forEach(el => el.onclick = () => deletePoint(el.dataset.deletePoint));

      document.querySelectorAll('[data-save]').forEach(el => el.onclick = () => savePoint(el.dataset.save === 'close' ? { closeAfter:true } : el.dataset.save === 'keep' ? { keepComponent:true } : {}));

      document.querySelectorAll('[data-close-dialog]').forEach(el => el.onclick = () => update(() => state.dialogs[el.dataset.closeDialog] = false));

      document.querySelectorAll('[data-field="search"]').forEach(el => el.oninput = () => update(() => state.search = el.value));

      document.querySelectorAll('[data-form]').forEach(el => el.oninput = () => update(() => state.form[el.dataset.form] = el.value));

      document.querySelectorAll('[data-project-form]').forEach(el => el.oninput = () => { state.projectForm[el.dataset.projectForm] = el.value; render(); });

      document.querySelectorAll('[data-structure-form]').forEach(el => el.oninput = () => { state.structureForm[el.dataset.structureForm] = el.value; render(); });

      const photoInput = document.getElementById('photo-input');

      if (photoInput) photoInput.onchange = () => update(() => { const files = Array.from(photoInput.files || []); state.form.photoAttached = files.length > 0; state.form.photoCount = files.length; state.form.photoNames = files.map(f => f.name || 'Foto'); });

    }


    render();
